function parseChordWikiPage(doc){ doc.querySelectorAll('script,style,nav,header,footer,.ad,iframe').forEach(el=>el.remove()); const bodyText=doc.body.textContent||''; let capo=0; const capoMatch=bodyText.match(/[Cc]apo[^0-9]*([0-9]+)/); if(capoMatch) capo=parseInt(capoMatch[1]); // Priority 1: textarea with ChordPro [chord] format const textarea=doc.querySelector('textarea'); if(textarea){ const src=textarea.value||textarea.textContent||''; if(src.includes('[')) return parseChordPro(src, capo); } // Priority 2: DOM-based extraction (ChordWiki renders chords as spans) const chordSpans=doc.querySelectorAll('[class*="chord"],[class*="Chord"]'); if(chordSpans.length>0){ return parseChordWikiDOM(doc, capo); } // Priority 3: pre/code block const pre=doc.querySelector('pre,code,.pre'); if(pre){ const txt=pre.textContent||''; if(txt.length>50){ if(txt.includes('[')) return parseChordPro(txt, capo); return parseRenderedChordWiki(txt, capo); } } // Priority 4: Full body text if(bodyText.includes('[')) return parseChordPro(bodyText, capo); return parseRenderedChordWiki(bodyText, capo); } function parseChordWikiDOM(doc, capo){ const sections=[]; let curSection={label:'Verse', lines:[]}; const chordArea=doc.querySelector('#chord,#chord_area,.chord_area,#main,pre')||doc.body; function walk(node){ if(node.nodeType===3){ const txt=(node.textContent||'').trim(); if(txt) return [{type:'text',value:txt}]; return []; } if(node.nodeType!==1) return []; const tag=node.tagName.toLowerCase(); const cls=(node.className||'').toLowerCase(); if(/^h[1-6]$/.test(tag)||cls.includes('section')||cls.includes('label')){ return [{type:'section',value:node.textContent.trim()}]; } if(cls.includes('chord')){ return [{type:'chord',value:node.textContent.trim()}]; } if(tag==='br'||tag==='hr') return [{type:'br'}]; if(/^(div|p|li|tr|td)$/.test(tag)){ const children=[].concat(...[...node.childNodes].map(walk)); return [...children,{type:'br'}]; } return [].concat(...[...node.childNodes].map(walk)); } const tokens=walk(chordArea); let chordBuf=[], lyricBuf=''; function flush(){ if(chordBuf.length>0||lyricBuf.trim()){ curSection.lines.push({chords:chordBuf.join(' '),lyric:lyricBuf.trim()}); chordBuf=[];lyricBuf=''; } } function pushSection(label){ if(curSection.lines.length>0) sections.push(curSection); curSection={label,lines:[]}; } for(const tok of tokens){ if(tok.type==='section'){ flush(); const lbl=tok.value; if(/(イントロ|intro)/i.test(lbl)) pushSection('イントロ'); else if(/(verse|aメロ)/i.test(lbl)) pushSection('Aメロ'); else if(/(chorus|サビ)/i.test(lbl)) pushSection('サビ'); else if(/(bridge|ブリッジ)/i.test(lbl)) pushSection('ブリッジ'); else if(/(outro|アウトロ)/i.test(lbl)) pushSection('アウトロ'); else if(lbl.length<20) pushSection(lbl); } else if(tok.type==='chord'){ if(lyricBuf.trim()&&chordBuf.length===0){ flush(); } chordBuf.push(tok.value); } else if(tok.type==='text'){ lyricBuf+=tok.value; } else if(tok.type==='br'){ flush(); } } flush(); if(curSection.lines.length>0) sections.push(curSection); if(sections.length===0||sections.every(s=>s.lines.length===0)){ return parseRenderedChordWiki(doc.body.textContent||'', capo); } return {capo, sections:sections.filter(s=>s.lines.length>0)}; }
